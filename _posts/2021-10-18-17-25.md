文章内容是用python写一个猜1到100的数字的小游戏，能储存玩家信息那种，不多废话了，直接开始。

```
name = input("请输入你的名字：")
# 输入用户名
f = open('game_data.txt', encoding='utf-8')
# 读取游戏信息txt文档，存储到f中
lines = f.readlines()
# 读取每一行内容并创建一个列表分行保存
f.close
# 关闭文件
datas = {}
创建一个叫datas的字典
for i in lines: 
# 遍历列表内容
    line = i.split()
# 把每一行信息的每个元素提取出来创建列表line
    datas[line[0]] = line[1:]
# 让列表第一个字符作为字典的key，剩下内容作为value
data = datas.get(name)
# 以name为key提取value内容
if data is None:
    data = [0, 0, 0, 0]
# 判断字典内容是否为空，为空则创建0，0，0，0的内容（0的含义后面写）
game_data = f.readline()
给game_data复制文件每一行内容
for data in game_data:
    data = game_data.split()
# 用data遍历game_data内容，并分别存储
game_times = int(data[0]) #游戏轮数
total_times = int(data[1]) #游戏总次数
min_times = int(data[2]) #最少游戏次数
avg_times = int(data[3]) #每轮游戏平均轮数
print("%s，\n游戏轮数为%d\n游戏总次数为%d\n最少%d次猜出结果\n平均%d次猜出结果" %(name, game_times, total_times, min_times, avg_times))
# 输出玩家信息

from random import randint
# 导入random库中randint函数
num = randint(1,100)
# 在1-100中随机取值赋给num
print("猜猜我在想什么数字？（1到100）")
# 不解释，太简单
A = True
B = True
# 方便后续操作
times = 0
# 记录每轮游戏初始次数，初始化为0
while A:
    guess = input()
    if str.isdigit(guess) == True:
#判断输入是否为数字，不为则退出程序，不保存数据
        if int(guess) > num:
            print("猜大了")
            times += 1
        elif int(guess) < num:
            print("猜小了")
            times += 1
        elif int(guess) == num:
            print("对，就是%d" %(num))
            A = False
            times += 1
            print("你猜了%d次" %times)
# 这段简单，不解释
    else:
        print("非法输入")
        A = False
        B = False

if B == True:
# 当所有输入都为数字时
    if game_times == 0 or times < min_times:
        min_times = times
    total_times +=times
    game_times += 1
    avg_times =int(total_times / game_times)
# 计算游戏轮数，游戏次数，总游戏次数，每轮游戏平均次数
    datas[name] = [str(game_times), str(total_times), str(min_times), str(avg_times)]
# 让字典指定key为指定value内容
    result = ''
# 指定一个空字符串用于存储数据
    for n in datas:
# 遍历字典内容
        line = n + ' ' + ' '.join(datas[n]) + '\n'
#让此行内容为字典value内容
        result += line
    f = open('game_data.txt','w')
    f.write(result)
    print("数据已保存")
else:
    print("没有数据被保存")
```
